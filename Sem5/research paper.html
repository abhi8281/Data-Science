Time Series Analysis:

In[]:
# Convert the 'created_at' column to a datetime object
df['created_at'] = pd.to_datetime(df['created_at'])
In[]:
# Resample data to analyze sentiment and tweet volume over time
sentiment_over_time = df.resample('D', on='created_at')['compound'].mean()
tweet_volume_over_time = df.resample('D', on='created_at')['original_text'].count()
In[]:
# Create time series plots
plt.figure(figsize=(12, 6))
plt.plot(sentiment_over_time, label='Sentiment', color='blue')
plt.xlabel('Date')
plt.ylabel('Mean Sentiment Score')
plt.title('Sentiment Over Time')
plt.legend(loc='upper right')

plt.figure(figsize=(12, 6))
plt.plot(tweet_volume_over_time, label='Tweet Volume', color='green')
plt.xlabel('Date')
plt.ylabel('Tweet Volume')
plt.title('Tweet Volume Over Time')
plt.legend(loc='upper right')
plt.show()
